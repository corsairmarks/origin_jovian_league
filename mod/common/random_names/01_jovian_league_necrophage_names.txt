### Jovian League Necrophages (duplicate of Necrophages from 00_empire_names.txt)

# Necrophage Hive Mind
empire_name_format = {
	random_weight = {
		factor = 0
		modifier = {
			add = 10
			OR = {
				has_government = "gov_hive_mind"
				has_government = "gov_parasitic_overmind"
			}
			has_origin = origin_jovian_league_necrophage
			is_pirate = no
			is_primitive = no
			is_fallen_empire = no
		}
	}
	format = "{<necrophagers_start> {[This.GetSpeciesAdj] {<hive_mind>}}}"
	adjective = "{[This.GetSpeciesAdj]}"
	prefix_format = "{[This.GetSpeciesAdj] {<hive_mind>}}"
	# Hungry Human Empire
}
